document.addEventListener('DOMContentLoaded', function() {
    // Create the notification bar element
    var notificationBar = document.createElement('div');
    notificationBar.setAttribute('id', 'notification-bar');
    notificationBar.textContent = 'Welcome to Pattikad Agritech! Stay tuned for important updates and messages.';

    // Optionally, add a close button
    var closeButton = document.createElement('button');
    closeButton.textContent = 'X';
    closeButton.style.float = 'right';
    closeButton.style.marginRight = '20px';
    closeButton.style.background = 'none';
    closeButton.style.border = 'none';
    closeButton.style.fontSize = '1rem';
    closeButton.style.color = '#333';
    
    closeButton.onclick = function() {
        notificationBar.style.display = 'none';
    };

    notificationBar.appendChild(closeButton);

    // Append the notification bar to the body
    document.body.insertBefore(notificationBar, document.body.firstChild);
});
